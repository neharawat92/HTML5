var w;

function VerifyPrimeno() {
  sessionStorage.inputNum = document.getElementById("num").value;

  if (typeof(Worker) !== "undefined") {
    if (typeof(w) == "undefined") {

      w = new Worker("web.js");
      w.postMessage(+document.getElementById("num").value);

    }
    w.onmessage = function(event) {
      document.getElementById("result").innerHTML = event.data;
    }
  } else {
    document.getElementById("result").innerHTML = "Error browser does not support the Web Workers";
  }
}

function ProcessHalt() {
  alert("Process ended");
  w.terminate();
  w = undefined;
}
